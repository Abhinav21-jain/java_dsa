n

0
1

.. first n fibonaccis
Constraints
1 <= n < 40
Sample Input
10
Sample Output
0
1

13
21
34

program
import java.util.*;
  
  public class Main{
  
  public static void main(String[] args) {
      // write your code here
      Scanner scn = new Scanner(System.in);
      int n = scn.nextInt();
      int a = 0;
      int b = 1;
      for (int i = 0; i < n; i++){
          System.out.println(a);
          int c = a + b;
          a = b;
          b = c;
      }
   }
